
CREATE TABLE users (
id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, 
email TEXT NOT NULL, 
hash TEXT NOT NULL, 
name TEXT NOT NULL,
year TEXT NOT NULL,
admin INTEGER NOT NULL
);

CREATE TABLE colleges (
collegeID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, 
name TEXT NOT NULL,
url TEXT NOT NULL,
urlworks INTEGER NOT NULL,
loginworks INTEGER NOT NULL
);

CREATE TABLE checklists (
detailID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, 
userID INTEGER NOT NULL,
collegeID INTEGER NOT NULL,
detail TEXT NOT NULL,
status TEXT NOT NULL,
date TEXT NOT NULL
);

CREATE TABLE logins (
loginID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, 
userID INTEGER NOT NULL,
collegeID INTEGER NOT NULL,
email INTEGER NOT NULL,
password TEXT NOT NULL,
changed INTEGER NOT NULL,
pageSource TEXT,
decisionOut INTEGER NOT NULL
);
